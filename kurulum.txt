# Proje dizinine geç
cd /var/www/cartridge

# Paketleri çek
npm install

# Önce prisma şemaları oluştur ve db push et
npx prisma generate
npx prisma db push

# Bir build et
npm run build

# PM2 ile sürekli ayakta kalmasını sağla
npm install -g pm2
pm2 start npm --name "cartridge" -- start

# NGINX yükle reverse proxy yapacağız
sudo apt install nginx -y

# Yeni bir konfigürasyon dosyası oluştur
sudo nano /etc/nginx/sites-available/cartridge

# İÇERİK
server {
    listen 80;
    server_name 192.168.0.213;

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}

# Dosyayı aktif edilen siteler arasına sembolik linkle ekle
sudo ln -s /etc/nginx/sites-available/cartridge /etc/nginx/sites-enabled/

# Varsayılan Nginx sayfasını kaldır
sudo rm /etc/nginx/sites-enabled/default

# Ayarları test et
sudo nginx -t

# Nginx'i yeniden başlat
sudo systemctl restart nginx
